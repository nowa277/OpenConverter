# OpenConverter Android v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a deliverable Android APK (arm64-v8a + x86_64) whose audio-conversion feature set matches the desktop Electron app — plaintext interconversion + NCM decryption priority — with Kotlin 1:1 ports of the 5 JS decoders and ffmpeg-kit for transcoding.

**Architecture:** Pure-Kotlin decoders (no NDK, no self-written C JNI) behind a registry; ffmpeg-kit full-gpl 6.0 invoked via its Java `FFmpegKit` API (command-line model, not libavcodec C API — this structurally avoids the v0.3.1 0-byte-FLAC bug); a JVM-testable `ConversionEngine` orchestrates the per-file loop behind facades; a Foreground Service runs conversions so backgrounding/lock-screen never kills a job; Spotify-inspired dark-only Material 3 Compose UI.

**Tech Stack:** Kotlin 1.9.24, AGP 8.5.2, Gradle 8.7, minSdk 24 / targetSdk 34 / compileSdk 34, Jetpack Compose BOM 2024.10.00, Material 3, ffmpeg-kit-full-gpl 6.0-2.LTS, javax.crypto (AES/RC4 in pure Kotlin), JUnit 4 + kotlinx-coroutines-test.

import org.junit.Assert.assertEquals
import org.junit.Assert.assertNotNull
import org.junit.Test

class DecoderContractTest {

    private class StubDecoder : Decoder {
        override val supportedExtensions = setOf(".stub")
        override fun decrypt(input: ByteArray) =
            DecryptResult(audio = byteArrayOf(1, 2, 3), format = "mp3")
    }

    @Test fun decrypt_returns_audio_and_format() {
        val r = StubDecoder().decrypt(byteArrayOf(0))
        assertEquals(listOf<Byte>(1, 2, 3), r.audio.toList())
        assertEquals("mp3", r.format)
        assertNotNull(r.audio)
    }

    @Test fun decryptResult_meta_defaults_null() {
        val r = StubDecoder().decrypt(byteArrayOf(0))
        assertEquals(null, r.meta)
    }

    val format: String,

package com.openconverter.app.decoders

/**
 * Sniff the container format of decrypted audio bytes by magic headers.
 * 1:1 port of src/decoders/kgm.js:inferFormat (the most complete JS sniffer).
 */
object FormatSniffer {
    // [0]=flac fLaC, [1]=mp3 ID3, [2]=ogg OggS, [3]=m4a ftyp@4, [4]=wav RIFF, [5]=aac, [6]=dff FRM8, [7]=wma ASF
    private val FLAC_HDR = byteArrayOf(0x66, 0x4c, 0x61, 0x43)
    private val MP3_HDR = byteArrayOf(0x49, 0x44, 0x33)
    private val OGG_HDR = byteArrayOf(0x4f, 0x67, 0x67, 0x53)
    private val WAV_HDR = byteArrayOf(0x52, 0x49, 0x46, 0x46)
    private val M4A_HDR = byteArrayOf(0x66, 0x74, 0x79, 0x70)
    private val WMA_HDR = byteArrayOf(
        0x30, 0x26, 0xb2.toByte(), 0x75, 0x8e.toByte(), 0x66, 0xcf.toByte(), 0x11,
        0xa6.toByte(), 0xd9.toByte(), 0x00, 0xaa.toByte(), 0x00, 0x62, 0xce.toByte(), 0x6c
    )
    private val DFF_HDR = byteArrayOf(0x46, 0x52, 0x4d, 0x38)

    fun sniff(b: ByteArray): String {
        if (b.size < 4) return "mp3"
        if (b[0] == MP3_HDR[0] && b[1] == MP3_HDR[1] && b[2] == MP3_HDR[2]) return "mp3"
        if (b[0] == FLAC_HDR[0] && b[1] == FLAC_HDR[1] && b[2] == FLAC_HDR[2] && b[3] == FLAC_HDR[3]) return "flac"
        if (b[0] == OGG_HDR[0] && b[1] == OGG_HDR[1] && b[2] == OGG_HDR[2] && b[3] == OGG_HDR[3]) return "ogg"
        if (b[0] == WAV_HDR[0] && b[1] == WAV_HDR[1] && b[2] == WAV_HDR[2] && b[3] == WAV_HDR[3]) return "wav"
        if (b[0].toInt() and 0xff == 0xff && (b[1].toInt() and 0xe0) == 0xe0) return "mp3" // MP3 frame sync
        if (b[0] == WMA_HDR[0] && b.size >= WMA_HDR.size) {
            var match = true
            for (i in WMA_HDR.indices) if (b[i] != WMA_HDR[i]) { match = false; break }
            if (match) return "wma"
        }
        if (b[0].toInt() and 0xff == 0xff && b[1].toInt() and 0xff == 0xf1) return "aac"
        if (b[0] == DFF_HDR[0] && b[1] == DFF_HDR[1] && b[2] == DFF_HDR[2] && b[3] == DFF_HDR[3]) return "dff"
        if (b.size >= 8 && b[4] == M4A_HDR[0] && b[5] == M4A_HDR[1] && b[6] == M4A_HDR[2] && b[7] == M4A_HDR[3]) return "m4a"
        return "mp3"
    }
}

 *
 * File layout:
 *   0x000..0x00A  Magic "yeelion-kuwo" (10 bytes)
 *   0x00A..0x010  6 bytes padding
 *   0x010..0x014  4-byte uint32 LE seed
 *   0x014..0x400  Reserved / padding
 *   0x400..end   Encrypted audio (XOR with derived 32-byte mask)
 *
 * Decryption:
 *   1. decimal = uint32_to_decimal_string(seed)
 *   2. mask first len(decimal) bytes = ASCII bytes of decimal
 *   3. mask remainder = 0
 *   4. for i in 0..32: mask[i] ^= ROOT[i]
 *   5. for i in 0..audio_len: audio[i] ^= mask[i % 32]
 *
 * Cipher is involutory (XOR is self-inverse).
 */
object KwmDecoder : Decoder {

    private val MAGIC = "yeelion-kuwo".toByteArray(Charsets.US_ASCII)
    private const val MAGIC_LEN = 10
    private val ROOT = "MoOtOITvINGwd2E6n0E1i7L5t2IoOoNk".toByteArray(Charsets.US_ASCII) // 32 bytes
    private const val MASK_SIZE = 32
    private const val AUDIO_OFFSET = 0x400
    private const val SEED_OFFSET = 0x10

    override val supportedExtensions: Set<String> = setOf(".kwm")

    private fun buildMask(seed: UInt): ByteArray {
        val decimal = seed.toString()
        val mask = ByteArray(MASK_SIZE)
        for (i in decimal.indices) mask[i] = decimal[i].code.toByte()
        for (i in 0 until MASK_SIZE) mask[i] = (mask[i].toInt() xor ROOT[i].toInt()).toByte()
        return mask
    }

    override fun decrypt(input: ByteArray): DecryptResult {
        require(input.size >= AUDIO_OFFSET) {
            "KWM: file too small (${input.size} < $AUDIO_OFFSET)"
        }
        for (i in 0 until MAGIC_LEN) {
            require(input[i] == MAGIC[i]) { "KWM: bad magic, expected \"yeelion-kuwo\"" }
        }
        // uint32 LE at SEED_OFFSET (mask off sign bit → UInt)
        val seed = (
            (input[SEED_OFFSET].toUByte().toUInt()) or
            (input[SEED_OFFSET + 1].toUByte().toUInt() shl 8) or
            (input[SEED_OFFSET + 2].toUByte().toUInt() shl 16) or
            (input[SEED_OFFSET + 3].toUByte().toUInt() shl 24)
        )
        val mask = buildMask(seed)
        val audio = input.copyOfRange(AUDIO_OFFSET, input.size)
        for (i in audio.indices) {
            audio[i] = (audio[i].toInt() xor mask[i % MASK_SIZE].toInt()).toByte()
        }
        return DecryptResult(audio = audio, format = FormatSniffer.sniff(audio))
    }
}

**Port notes (read these — they are the JS→Kotlin pitfalls for this file):**
- `Buffer.readUInt32LE` → manual LE assemble into `UInt`. Do NOT use `Int` then mask; the unsigned widening via `toUByte().toUInt()` avoids sign-extension bugs.
- `Buffer.alloc(n)` zero-fills; `ByteArray(n)` also zero-fills in Kotlin — same semantics.
- `seed.toString(10)` (JS, radix 10) → `seed.toString()` (Kotlin UInt default is decimal). Verify parity test catches any mismatch.
- The `xor` chain stays in `Int` and narrows with `.toByte()` at the end — matches JS `^` on bytes (JS treats them as int32 then assigns back as uint8).
- Format sniffing happens INSIDE the decoder (mirrors `kwm.js:decodeFile` calling `inferFormat`), so the engine sees a non-null `format` from `Decoder.decrypt`.

            value = value xor TABLE2[offset % 272].toInt()
            offset = offset shr 4
        }
        return (MASK_V2_PRE_DEF[pos % 272].toInt() xor value).toByte()
    }

    override fun decrypt(input: ByteArray): DecryptResult {
        require(input.size >= 0x2C) { "KGM/VPR: file too small (${input.size} < 0x2C)" }
        val isVpr: Boolean = when {
            input.copyOfRange(0, 16).contentEquals(KGM_HEADER) -> false
            input.copyOfRange(0, 16).contentEquals(VPR_HEADER) -> true
            else -> throw IllegalArgumentException("KGM/VPR: bad magic")
        }
        val headerLen = (
            (input[0x10].toUByte().toInt()) or
            (input[0x11].toUByte().toInt() shl 8) or
            (input[0x12].toUByte().toInt() shl 16) or
            (input[0x13].toUByte().toInt() shl 24)
        )
        require(headerLen in 0x2C..input.size) { "KGM/VPR: invalid headerLen $headerLen" }

        val key = ByteArray(17)
        System.arraycopy(input, 0x1C, key, 0, 16)
        // key[16] is implicitly 0 from ByteArray init.

        val audio = input.copyOfRange(headerLen, input.size)
        for (i in audio.indices) {
            var med8 = (key[i % 17].toInt() xor audio[i].toInt()) and 0xff
            med8 = med8 xor ((med8 and 0x0f) shl 4)
            var msk8 = getMask(i).toInt() and 0xff
            msk8 = msk8 xor ((msk8 and 0x0f) shl 4)
            var b = (med8 xor msk8) and 0xff
            if (isVpr) b = b xor (VPR_MASK_DIFF[i % 17].toInt() and 0xff)
            audio[i] = b.toByte()
        }
        return DecryptResult(audio = audio, format = FormatSniffer.sniff(audio))
    }
}

/**
 * Pure Kotlin port of src/decoders/qmc.js — QMCv1 portion (headerless XOR with
 * an 8x7 state machine, no key needed). v2 (mflac/mgg/bkc, requires user-supplied

object QmcDecoder : Decoder {
    // v1 only — v2 extensions deliberately omitted until decryption is wired.
    override val supportedExtensions: Set<String> = setOf(
        ".qmc0", ".qmc3", ".qmcflac", ".qmcogg", ".qmc1", ".qmc2", ".tkm"
    )

    private val SEED_MAP: Array<ByteArray> = arrayOf(
        byteArrayOf(0x4a, 0xd6.toByte(), 0xca.toByte(), 0x90.toByte(), 0x67, 0xf7.toByte(), 0x52),
        byteArrayOf(0x5e, 0x95.toByte(), 0x23, 0x9f.toByte(), 0x13, 0x11, 0x7e),
        byteArrayOf(0x47, 0x74, 0x3d, 0x90.toByte(), 0xaa.toByte(), 0x3f, 0x51),
        byteArrayOf(0xc6.toByte(), 0x09, 0xd5.toByte(), 0x9f.toByte(), 0xfa.toByte(), 0x66, 0xf9.toByte()),
        byteArrayOf(0xf3.toByte(), 0xd6.toByte(), 0xa1.toByte(), 0x90.toByte(), 0xa0.toByte(), 0xf7.toByte(), 0xf0.toByte()),
        byteArrayOf(0x1d, 0x95.toByte(), 0xde.toByte(), 0x9f.toByte(), 0x84.toByte(), 0x11, 0xf4.toByte()),
        byteArrayOf(0x0e, 0x74, 0xbb.toByte(), 0x90.toByte(), 0xbc.toByte(), 0x3f, 0x92.toByte()),
        byteArrayOf(0x00, 0x09, 0x5b, 0x9f.toByte(), 0x62, 0x66, 0xa1.toByte()),
    )

    /** State machine — 1:1 with src/decoders/qmc.js QmcSeed. */
    private class QmcSeed {
        var x = -1; var y = 8; var dx = 1; var index = -1
        fun nextMask(): Int {
            index++
            val ret = when {
                x < 0 -> { dx = 1; y = (8 - y) % 8; 0xc3 }
                x > 6 -> { dx = -1; y = 7 - y; 0xd8 }
                else  -> SEED_MAP[y][x].toInt() and 0xff
            }
            x += dx
            if (index == 0x8000 || (index > 0x8000 && (index + 1) % 0x8000 == 0)) {
                return nextMask()
            }
            return ret
        }
    }

    override fun decrypt(input: ByteArray): DecryptResult {
        val seed = QmcSeed()
        val out = ByteArray(input.size)
        for (i in input.indices) {
            out[i] = ((input[i].toInt() and 0xff) xor seed.nextMask()).toByte()
        }
        return DecryptResult(audio = out, format = FormatSniffer.sniff(out))
    }

    // ---- v2 reserved (DO NOT enable in v1) ----
    // QMCv2 (mflac/mgg/bkc*) needs a 128-byte working key derived from a
    // user-provided ekey via key_compress. Algorithm in src/decoders/qmc.js
    // (decryptV2Buffer + keyCompress + shiftMix). Adding v2:
    //   1. Restore those 3 functions here as private fun.
    //   2. Add v2 extensions to supportedExtensions.
    //   3. Override decrypt(input: ByteArray, ekey: ByteArray?) — needs an

    //   4. Wire ekey UI in SettingsScreen + persist via DataStore.
    //   5. Add parity test using tests/gen-qmc-v2-synthetic.js fixture.
}

```kotlin
package com.openconverter.app.decoders

import javax.crypto.Cipher
import javax.crypto.spec.SecretKeySpec

/**
 * Pure Kotlin port of src/decoders/ncm.js — NetEase Cloud Music NCM container.
 *
 * Layout:
 *   0..8    magic "CTENFDAM"
 *   8..10   gap
 *   10..14  uint32 LE keyLength
 *   14..    encrypted RC4 key (XOR 0x64 → AES-128-ECB(CORE_KEY) → unpad → strip "neteasecloudmusic" prefix)
 *   ..      uint32 LE metaLength + meta (XOR 0x63 → b64 → AES-128-ECB(META_KEY) → unpad → strip "music:" prefix → JSON)
 *   ..      5-byte gap
 *   ..      uint32 LE imageSpace, uint32 LE imageSize, image bytes, padding
 *   ..      audio body (RC4 keystream from S-box)
 *
 * The audio cipher is the modified RC4 from ncmdump: keystream byte for data[i]
 * is k[(i+1)%256] where k[i] = S[(S[i] + S[(i + S[i]) & 0xff]) & 0xff].
 */
object NcmDecoder : Decoder {
    override val supportedExtensions: Set<String> = setOf(".ncm")

    private val MAGIC = "CTENFDAM".toByteArray(Charsets.US_ASCII)
    private val CORE_KEY = hexToBytes("687A4852416D736F356B496E62617857") // "hzHRAmso5kInbaxW"
    private val META_KEY = hexToBytes("2331346C6A6B5F215C5D2630553C2728") // "#14ljk_!\\]&0U<'("
    private const val PREFIX_LEN = 17 // "neteasecloudmusic" + "\0"

    override fun decrypt(input: ByteArray): DecryptResult {
        require(input.size >= 10 && input.copyOfRange(0, 8).contentEquals(MAGIC)) {
            "NCM: missing CTENFDAM magic"
        }
        var off = 10
        val keyLen = readUInt32LE(input, off); off += 4
        require(keyLen in 1..input.size) { "NCM: invalid key length $keyLen" }
        val keyEnc = input.copyOfRange(off, off + keyLen); off += keyLen
        for (i in keyEnc.indices) keyEnc[i] = (keyEnc[i].toInt() xor 0x64).toByte()
        val keyPlain = pkcs7Unpad(aesEcbDecrypt(keyEnc, CORE_KEY))
        require(keyPlain.size >= PREFIX_LEN) { "NCM: key block too short" }
        val rc4Key = keyPlain.copyOfRange(PREFIX_LEN, keyPlain.size)
        val sBox = buildRc4Sbox(rc4Key)

        val metaLen = readUInt32LE(input, off); off += 4
        var metaJson: String? = null
        if (metaLen > 0) {
            require(off + metaLen <= input.size) { "NCM: meta length exceeds file" }
            val metaEnc = input.copyOfRange(off, off + metaLen); off += metaLen
            for (i in metaEnc.indices) metaEnc[i] = (metaEnc[i].toInt() xor 0x63).toByte()
            val b64 = String(metaEnc.copyOfRange(22, metaEnc.size), Charsets.UTF_8)
            try {
                val decoded = java.util.Base64.getDecoder().decode(b64)
                val plain = pkcs7Unpad(aesEcbDecrypt(decoded, META_KEY))
                if (plain.size > 6) metaJson = String(plain.copyOfRange(6, plain.size), Charsets.UTF_8)
            } catch (_: Exception) { /* meta is best-effort */ }
        }

        off += 5 // gap
        val imageSpace = readUInt32LE(input, off); off += 4
        val imageSize  = readUInt32LE(input, off); off += 4
        off += imageSpace // covers imageSize bytes + (imageSpace - imageSize) padding
        require(off <= input.size) { "NCM: header parsing past EOF" }

        val audioEnc = input.copyOfRange(off, input.size)
        val audio = rc4Decrypt(sBox, audioEnc)
        return DecryptResult(audio = audio, format = FormatSniffer.sniff(audio), meta = metaJson)
    }

    // ---- helpers ----

    private fun aesEcbDecrypt(block: ByteArray, key: ByteArray): ByteArray {
        val c = Cipher.getInstance("AES/ECB/NoPadding")
        c.init(Cipher.DECRYPT_MODE, SecretKeySpec(key, "AES"))
        return c.doFinal(block)
    }

    private fun pkcs7Unpad(b: ByteArray): ByteArray {
        require(b.isNotEmpty()) { "PKCS7 unpad: empty" }
        val pad = b.last().toInt() and 0xff
        require(pad in 1..16) { "PKCS7 unpad: invalid pad $pad" }
        return b.copyOfRange(0, b.size - pad)
    }

    private fun buildRc4Sbox(key: ByteArray): IntArray {
        val s = IntArray(256) { it }
        var j = 0
        for (i in 0 until 256) {
            j = (j + s[i] + (key[i % key.size].toInt() and 0xff)) and 0xff
            val t = s[i]; s[i] = s[j]; s[j] = t
        }
        return s
    }

    private fun rc4Decrypt(s: IntArray, data: ByteArray): ByteArray {
        // Modified RC4: keystream byte = k[(i+1)%256] where
        //   k[i] = s[(s[i] + s[(i + s[i]) & 0xff]) & 0xff]
        val k = IntArray(256)
        for (i in 0 until 256) k[i] = s[(s[i] + s[(i + s[i]) and 0xff]) and 0xff]
        val out = ByteArray(data.size)
        for (i in data.indices) out[i] = ((data[i].toInt() and 0xff) xor k[(i + 1) and 0xff]).toByte()
        return out
    }

    private fun readUInt32LE(b: ByteArray, off: Int): Int =
        (b[off].toInt() and 0xff) or
            ((b[off+1].toInt() and 0xff) shl 8) or
            ((b[off+2].toInt() and 0xff) shl 16) or
            ((b[off+3].toInt() and 0xff) shl 24)

    private fun hexToBytes(hex: String): ByteArray {
        val out = ByteArray(hex.length / 2)
        for (i in out.indices) {
            out[i] = ((hex[i*2].digitToInt(16) shl 4) or hex[i*2+1].digitToInt(16)).toByte()
        }
        return out
    }
}

package com.openconverter.app.ffmpeg

/**
 * Abstraction over ffmpeg-kit so the engine can be unit-tested with a fake.
 * Only one production impl: [FfmpegKitRunner]. Tests use FakeFfmpegRunner.
 */
interface FfmpegRunner {
    /**
     * Run ffmpeg synchronously (suspending). Reports progress 0..100 as it
     * advances; final 100 is emitted on success.
     *
     * @param input absolute filesystem path
     * @param output absolute filesystem path (must NOT equal input)
     * @param format one of "mp3"|"flac"|"wav"|"m4a"|"ogg"
     * @param bitrate e.g. "320k"; null = codec default
     */
    suspend fun execute(
        input: String,
        output: String,
        format: String,
        bitrate: String?,
        onProgress: (percent: Int) -> Unit = {},
    ): Result<Unit>
}

/**
 * Pure command-line argv builder. Mirrors src/main/ffmpeg.js flag selection so
 * Android and desktop transcode behavior stay aligned.
 *
 * Codec choices are explicit (no auto-detect) for reproducibility: the same
 * input + format always yields the same codec and the same args (modulo bitrate).
 */
object FfmpegArgs {
    fun build(input: String, output: String, format: String, bitrate: String?): List<String> {
        require(input != output) { "ffmpeg refuses input==output; engine must use a temp output path" }
        val codec = codecFor(format)
        val args = mutableListOf("-y", "-i", input, "-codec:a", codec)
        if (bitrate != null && format != "wav" && format != "flac") {
            args += listOf("-b:a", bitrate)
        }
        args += output
        return args
    }

    private fun codecFor(format: String): String = when (format.lowercase()) {
        "mp3"  -> "libmp3lame"
        "flac" -> "flac"
        "wav"  -> "pcm_s16le"
        "m4a"  -> "aac"
        "ogg"  -> "libvorbis"
        else   -> throw IllegalArgumentException("Unsupported target format: $format")
    }
}
```

- [ ] **Step 1: Write `FfmpegKitRunner.kt`**

```kotlin
package com.openconverter.app.ffmpeg

import com.arthenica.ffmpegkit.FFmpegKit
import com.arthenica.ffmpegkit.FFmpegKitConfig
import com.arthenica.ffmpegkit.ReturnCode
import com.arthenica.ffmpegkit.Statistics
import com.arthenica.ffmpegkit.StatisticsCallback
import kotlinx.coroutines.suspendCancellableCoroutine
import kotlin.coroutines.resume

/**
 * Production impl. Each `execute` call:
 *   1. installs a process-global StatisticsCallback that translates ffmpeg's
 *      time= log lines into 0..100 progress (best-effort; ffmpeg-kit doesn't
 *      know total duration, so we approximate against an upper bound the
 *      caller can override later).
 *   2. runs FFmpegKit.executeAsync with the args from FfmpegArgs.build.
 *   3. resumes the coroutine with success/failure based on the return code.
 *   4. invokeOnCancellation cancels the ffmpeg session.
 *
 * Notes on progress: ffmpeg-kit emits stats with `getTime()` in milliseconds.
 * Without a probed duration, we report monotonic progress capped at 99 until
 * completion (then 100). Probing total duration is the engine's job (Milestone 5).
 */
class FfmpegKitRunner : FfmpegRunner {

    @Volatile private var totalDurationMs: Long = 0L

    /** Called by the engine BEFORE execute() if it knows the duration. */
    fun setTotalDurationMs(ms: Long) { totalDurationMs = ms.coerceAtLeast(0L) }

    override suspend fun execute(
        input: String,
        output: String,
        format: String,
        bitrate: String?,
        onProgress: (percent: Int) -> Unit,
    ): Result<Unit> = suspendCancellableCoroutine { cont ->
        val args = FfmpegArgs.build(input, output, format, bitrate).joinToString(" ") {
            // ffmpeg-kit takes a single command string; quote args containing spaces.
            if (it.contains(' ')) "\"$it\"" else it
        }

        val statsCallback = StatisticsCallback { stats: Statistics ->
            val total = totalDurationMs
            val pct = if (total > 0) {
                val p = (stats.time * 100L / total).toInt()
                p.coerceIn(0, 99)
            } else {
                // No probed duration: report a heartbeat (50%) until completion.
                50
            }
            onProgress(pct)
        }
        FFmpegKitConfig.enableStatisticsCallback(statsCallback)

        val session = FFmpegKit.executeAsync(args) { sess ->
            FFmpegKitConfig.enableStatisticsCallback(null)
            val rc = sess.returnCode
            if (ReturnCode.isSuccess(rc)) {
                onProgress(100)
                cont.resume(Result.success(Unit))
            } else if (ReturnCode.isCancel(rc)) {
                cont.resume(Result.failure(InterruptedException("ffmpeg cancelled")))
            } else {
                cont.resume(Result.failure(RuntimeException("ffmpeg rc=${rc} \n${sess.allLogsAsString.takeLast(2000)}")))
            }
        }

        cont.invokeOnCancellation {
            FFmpegKit.cancel(session.sessionId)
            FFmpegKitConfig.enableStatisticsCallback(null)
        }
    }
}
```
